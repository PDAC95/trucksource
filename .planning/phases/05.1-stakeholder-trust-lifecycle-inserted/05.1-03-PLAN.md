renewListing pattern (5.1-RESEARCH Example 7) — owner-scoped, zero-rows→not_found:
```typescript
export async function renewListing(id: number) {
  const supabase = await createClient();
  const { data: claims } = await supabase.auth.getClaims();   // getClaims, never getSession
  const userId = claims?.claims?.sub;
  if (!userId) return { ok: false, error: "unauthenticated" } as const;
  const { data, error } = await supabase.from("listings")
    .update({ expires_at: new Date(Date.now() + 90*864e5).toISOString() })
    .eq("id", id).eq("seller_id", userId).eq("status", "active")   // ACTIVE only
    .select("id, expires_at");
  if (error) return { ok: false, error: "invalid" } as const;
  if (!data?.length) return { ok: false, error: "not_found" } as const;
  return { ok: true, expiresAt: data[0].expires_at } as const;
}
// reactivateListing: same but .eq("status","expired") and .update({ status:'active', expires_at })
```
